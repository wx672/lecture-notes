        movl %ecx, page_pde_offset(%edx)  # Store kernel PDE entry

        movl $1024, %ecx
11:
        stosl   # movl %eax, (%edi)

        addl $0x1000, %eax
        loop 11b
        # End condition: we must map up to and including INIT_MAP_BEYOND_END

        # attribute bits 
        leal (INIT_MAP_BEYOND_END + 0x007)(%edi), %ebp
        cmpl %ebp, %eax
        jb 10b
        movl %edi, (init_pg_tables_end - __PAGE_OFFSET)
