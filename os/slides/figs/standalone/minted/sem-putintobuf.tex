void producer() {
  while (true) {
    item = produceItem();
    down(spaces);
    putIntoBuffer(item);
    up(items);
  }
}

\begin{ccode}
void consumer() {
  while (true) {
    down(items);
    item = rmFromBuffer();
    up(spaces);
    consumeItem(item);
  }
}
